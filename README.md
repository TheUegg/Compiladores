# Compiladores

## Gramática:
```
PROGRAM -> STATEMENT PROGRAM' | FUNCLIST PROGRAM' | ε
PROGRAM' -> STATEMENT PROGRAM' | ε
FUNCLIST -> FUNCDEF FUNCLIST'
FUNCLIST' -> FUNCDEF FUNCLIST' | ε
FUNCDEF -> def ident ( PARAMLIST ) { STATELIST }
PARAMLIST -> int ident PARAMLIST' | float ident PARAMLIST' | string ident PARAMLIST' | tint ident PARAMLIST' | tfloat ident PARAMLIST' | tstring ident PARAMLIST'
PARAMLIST' -> , PARAMLIST | ε
STATEMENT -> VARDECL ; | ATRIBSTAT ; | PRINTSTAT ; | READSTAT ; | RETURNSTAT ; | IFSTAT | FORSTAT | { STATELIST } | break ; | ;
VARDECL -> int ident SELECTINDEX | float ident SELECTINDEX | string ident SELECTINDEX | tint ident [ int_constant ] | tfloat ident [ int_constant ] | tstring ident [ int_constant ]
SELECTINDEX -> [ int_constant ] SELECTINDEX' | ε
SELECTINDEX' -> [ int_constant ] SELECTINDEX' | ε
ATRIBSTAT -> LVALUE EQUAL
EQUAL -> = EQUAL'
EQUAL' -> BOOLEXPRESSION | ALLOCEXPRESSION | FUNCCALL
LVALUE -> ident [ NUMEXPRESSION ]
FUNCCALL -> ident( FUNCCALL'
FUNCALL -> PARAMLISTCALL ) | )
PARAMLISTCALL -> ident PARAMLISTCALL'
PARAMLISTCALL' -> , PARAMLISTCALL | ε
PRINTSTAT -> print BOOLEXPRESSION
READSTAT -> read LVALUE
RETURNSTAT -> return
IFSTAT -> if ( BOOLEXPRESSION ) STATEMENT ELSE
ELSE -> else STATEMENT | ε
FORSTAT -> for ( ATRIBSTAT ; EXPRESSION ; ATRIBSTAT ) STATEMENT
STATELIST -> STATEMENT STATELIST' | ε
STATELIST' -> STATELIST | ε
ALLOCEXPRESSION -> new int NUMEXPRESSION2 | float NUMEXPRESSION2 | string NUMEXPRESSION2
NUMEXPRESSION2 -> [ NUMEXPRESSION2'
NUMEXPRESSION2 -> NUMEXPRESSION ] NUMEXPRESSION2 | ]
BOOLEXPRESSION -> EXPRESSION BOOLEXPRESSION' | ( BOOLEXPRESSION ) | not BOOLEXPRESSION
BOOLEXPRESSION' -> or BOOLEXPRESSION | and BOOLEXPRESSION | ε
EXPRESSION -> NUMEXPRESSION EXPRESSION'
EXPRESSION' -> < NUMEXPRESSION | > NUMEXPRESSION | < = NUMEXPRESSION | > = NUMEXPRESSION | = = NUMEXPRESSION | ! = NUMEXPRESSION | ε
NUMEXPRESSION -> TERM NUMEXPRESSION'
NUMEXPRESSION' -> + TERM | - TERM | ε
TERM -> UNARYEXPR TERM'
TERM' -> * UNARYEXPR | / UNARYEXPR | % UNARYEXPR | ε
UNARYEXPR -> FACTOR UNARYEXPR'
UNARYEXPR' -> + FACTOR | - FACTOR | ε
FACTOR -> int constant | float constant | string constant | null | LVALUE | ( NUMEXPRESSION )
```
